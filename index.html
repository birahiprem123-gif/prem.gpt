/** * NEXUS Dashboard — Login Apps Script (with Status/Active-Disable check) * ───────────────────────────────────────────────────────────────────── * Deploy this as a Web App: * Deploy → New deployment → type: Web app * Execute as: Me * Who has access: Anyone ← MUST be "Anyone" * * Matches your "NEXUS Database" sheet, tab "users", columns: * Username | Password | Name | Branch | Role | Status * prem | 123456 | Narayan Pokhrel | SHUKLAGANDAKI | Admin | Active * sagar | sagar51 | Sagar dhakal | Syangja | editter| Active * amrit | 123456 | Amrit pangnii | Dumre | editter| Disable * * The dashboard calls: * GET ?action=login&username=...&password=... * and expects back one of: * Success: {"success": true, "name": "...", "branch": "...", "role": "..."} * Failure: {"success": false, "message": "..."} */ const USERS_SHEET_NAME = 'users'; // matches your tab name exactly (lowercase 'users') function doGet(e) { const action = (e.parameter.action || '').toLowerCase(); if (action === 'login') { return handleLogin(e); } return jsonResponse({ success: false, message: 'Unknown action: ' + action }); } function handleLogin(e) { const username = (e.parameter.username || '').trim(); const password = (e.parameter.password || '').trim(); if (!username || !password) { return jsonResponse({ success: false, message: 'Username and password are required' }); } const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(USERS_SHEET_NAME); if (!sheet) { return jsonResponse({ success: false, message: 'Server misconfigured: "' + USERS_SHEET_NAME + '" sheet not found' }); } const data = sheet.getDataRange().getValues(); const headers = data[0].map(h => String(h).trim().toLowerCase()); const colUsername = headers.indexOf('username'); const colPassword = headers.indexOf('password'); const colName = headers.indexOf('name'); const colBranch = headers.indexOf('branch'); const colRole = headers.indexOf('role'); // Accept either "status" or "st" as the header name for the active/disable column const colStatus = headers.indexOf('status') !== -1 ? headers.indexOf('status') : headers.indexOf('st'); if (colUsername === -1 || colPassword === -1) { return jsonResponse({ success: false, message: 'Server misconfigured: missing Username/Password columns' }); } for (let i = 1; i < data.length; i++) { const row = data[i]; const rowUser = String(row[colUsername] || '').trim(); const rowPass = String(row[colPassword] || '').trim(); if (rowUser.toLowerCase() === username.toLowerCase() && rowPass === password) { // ── Status / Active-Disable check ────────────────────────── if (colStatus !== -1) { const status = String(row[colStatus] || '').trim().toLowerCase(); if (status === 'disable' || status === 'disabled' || status === 'inactive') { return jsonResponse({ success: false, message: 'Yo account disable garieko cha. Admin lai sampark garnus.' }); } } return jsonResponse({ success: true, name: colName !== -1 ? String(row[colName] || rowUser) : rowUser, branch: colBranch !== -1 ? String(row[colBranch] || '') : '', role: colRole !== -1 ? String(row[colRole] || 'user') : 'user' }); } } return jsonResponse({ success: false, message: 'Incorrect username or password' }); } function jsonResponse(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
